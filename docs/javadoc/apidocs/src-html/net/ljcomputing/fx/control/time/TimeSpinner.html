017package net.ljcomputing.fx.control.time;
018
019import java.time.LocalDate;
020import java.time.LocalDateTime;
021import java.time.LocalTime;
022import java.time.format.DateTimeFormatter;
023
024import javafx.beans.property.ObjectProperty;
025import javafx.beans.property.SimpleObjectProperty;
026import javafx.scene.control.Spinner;
027import javafx.scene.control.SpinnerValueFactory;
028import javafx.scene.control.TextFormatter;
029import javafx.scene.input.InputEvent;
030import javafx.util.StringConverter;
031
032/**
033 * A time spinner.
034 * <p>
035 * Original code: http://stackoverflow.com/questions/32613619/how-to-make-a-timespinner-in-javafx/32617768#32617768
036 * 
037 * @author James G. Willmore
038 *
039 */
040public class TimeSpinner extends Spinner<LocalTime> implements Comparable<Spinner<LocalTime>> {
041  
042  /** The formatter. */
043  private final DateTimeFormatter formatter = DateTimeFormatter.ofPattern("H:mm:ss");
044  
045  /**
046   * Mode represents the unit that is currently being edited.
047   * For convenience expose methods for incrementing and decrementing that
048   * unit, and for selecting the appropriate portion in a spinner's editor.
049   */
050  enum Mode {
051
052    HOURS {
053
054      /**
055       * @see net.ljcomputing.fx.control.time.TimeSpinner.Mode#increment(java.time.LocalTime, int)
056       */
057      @Override
058      LocalTime increment(LocalTime time, int steps) {
059        return time.plusHours(steps);
060      }
061
062      /**
063       * @see net.ljcomputing.fx.control.time.TimeSpinner.Mode#select(net.ljcomputing.fx.control.time.TimeSpinner)
064       */
065      @Override
066      void select(TimeSpinner spinner) {
067        int index = spinner.getEditor().getText().indexOf(':');
068        spinner.getEditor().selectRange(0, index);
069      }
070    },
071    MINUTES {
072
073      /**
074       * @see net.ljcomputing.fx.control.time.TimeSpinner.Mode#increment(java.time.LocalTime, int)
075       */
076      @Override
077      LocalTime increment(LocalTime time, int steps) {
078        return time.plusMinutes(steps);
079      }
080
081      /**
082       * @see net.ljcomputing.fx.control.time.TimeSpinner.Mode#select(net.ljcomputing.fx.control.time.TimeSpinner)
083       */
084      @Override
085      void select(TimeSpinner spinner) {
086        int hrIndex = spinner.getEditor().getText().indexOf(':');
087        int minIndex = spinner.getEditor().getText().indexOf(':', hrIndex + 1);
088        spinner.getEditor().selectRange(hrIndex + 1, minIndex);
089      }
090    },
091    SECONDS {
092
093      /**
094       * @see net.ljcomputing.fx.control.time.TimeSpinner.Mode#increment(java.time.LocalTime, int)
095       */
096      @Override
097      LocalTime increment(LocalTime time, int steps) {
098        return time.plusSeconds(steps);
099      }
100
101      /**
102       * @see net.ljcomputing.fx.control.time.TimeSpinner.Mode#select(net.ljcomputing.fx.control.time.TimeSpinner)
103       */
104      @Override
105      void select(TimeSpinner spinner) {
106        int index = spinner.getEditor().getText().lastIndexOf(':');
107        spinner.getEditor().selectRange(index + 1,
108            spinner.getEditor().getText().length());
109      }
110    };
111
112    /**
113     * Increment.
114     *
115     * @param time the time
116     * @param steps the steps
117     * @return the local time
118     */
119    abstract LocalTime increment(LocalTime time, int steps);
120
121    /**
122     * Select.
123     *
124     * @param spinner the spinner
125     */
126    abstract void select(TimeSpinner spinner);
127
128    /**
129     * Decrement.
130     *
131     * @param time the time
132     * @param steps the steps
133     * @return the local time
134     */
135    LocalTime decrement(LocalTime time, int steps) {
136      return increment(time, -steps);
137    }
138  }
139
140  // Property containing the current editing mode:
141
142  /** The mode. */
143  private final ObjectProperty<Mode> mode = new SimpleObjectProperty<>(
144      Mode.HOURS);
145
146  /**
147   * Mode property.
148   *
149   * @return the object property
150   */
151      public ObjectProperty<Mode> modeProperty() {
152    return mode;
153  }
154
155  /**
156   * Gets the mode.
157   *
158   * @return the mode
159   */
160  public final Mode getMode() {
161    return modeProperty().get();
162  }
163
164  /**
165   * Sets the mode.
166   *
167   * @param mode the new mode
168   */
169  public final void setMode(Mode mode) {
170    modeProperty().set(mode);
171  }
172
173  /**
174   * Instantiates a new time spinner.
175   *
176   * @param time the time
177   */
178  public TimeSpinner(LocalTime time) {
179    setEditable(true);
180
181    // Create a StringConverter for converting between the text in the
182    // editor and the actual value:
183
184    
185
186    StringConverter<LocalTime> localTimeConverter = new StringConverter<LocalTime>() {
187
188      @Override
189      public String toString(LocalTime time) {
190        if(null != time) {
191          return formatter.format(time);
192        }
193        
194        return null;
195      }
196
197      @Override
198      public LocalTime fromString(String string) {
199        String[] tokens = string.split(":");
200        int hours = getIntField(tokens, 0);
201        int minutes = getIntField(tokens, 1);
202        int seconds = getIntField(tokens, 2);
203        int totalSeconds = (hours * 60 + minutes) * 60 + seconds;
204        return LocalTime.of((totalSeconds / 3600) % 24,
205            (totalSeconds / 60) % 60, seconds % 60);
206      }
207
208      private int getIntField(String[] tokens, int index) {
209        if (tokens.length <= index || tokens[index].isEmpty()) {
210          return 0;
211        }
212        return Integer.parseInt(tokens[index]);
213      }
214
215    };
216
217    // The textFormatter both manages the text <-> LocalTime conversion,
218    // and vetoes any edits that are not valid. We just make sure we have
219    // two colons and only digits in between:
220
221    TextFormatter<LocalTime> textFormatter = new TextFormatter<LocalTime>(
222        localTimeConverter, LocalTime.now(), c -> {
223          String newText = c.getControlNewText();
224
225          if (newText.matches("[0-9]{0,2}:[0-9]{0,2}:[0-9]{0,2}")) {
226            return c;
227          }
228
229          return null;
230        });
231
232    // The spinner value factory defines increment and decrement by
233    // delegating to the current editing mode:
234
235    SpinnerValueFactory<LocalTime> valueFactory = new SpinnerValueFactory<LocalTime>() {
236
237      {
238
239        setConverter(localTimeConverter);
240        setValue(time);
241      }
242
243      @Override
244      public void decrement(int steps) {
245        setValue(mode.get().decrement(getValue(), steps));
246        mode.get().select(TimeSpinner.this);
247      }
248
249      @Override
250      public void increment(int steps) {
251        setValue(mode.get().increment(getValue(), steps));
252        mode.get().select(TimeSpinner.this);
253      }
254
255    };
256
257    this.setValueFactory(valueFactory);
258    this.getEditor().setTextFormatter(textFormatter);
259
260    // Update the mode when the user interacts with the editor.
261    // This is a bit of a hack, e.g. calling spinner.getEditor().positionCaret()
262    // could result in incorrect state. Directly observing the caretPostion
263    // didn't work well though; getting that to work properly might be
264    // a better approach in the long run.
265    this.getEditor().addEventHandler(InputEvent.ANY, e -> {
266      int caretPos = this.getEditor().getCaretPosition();
267      int hrIndex = this.getEditor().getText().indexOf(':');
268      int minIndex = this.getEditor().getText().indexOf(':', hrIndex + 1);
269      if (caretPos <= hrIndex) {
270        mode.set(Mode.HOURS);
271      } else if (caretPos <= minIndex) {
272        mode.set(Mode.MINUTES);
273      } else {
274        mode.set(Mode.SECONDS);
275      }
276    });
277
278    // When the mode changes, select the new portion:
279    mode.addListener((obs, oldMode, newMode) -> newMode.select(this));
280
281  }
282
283  /**
284   * Instantiates a new time spinner.
285   */
286  public TimeSpinner() {
287    this(LocalTime.now());
288  }
289
290  /**
291   * Sets the local time.
292   *
293   * @param time the new local time
294   */
295  public void setLocalTime(LocalTime time) {
296    this.getValueFactory().setValue(time);
297  }
298  
299  /**
300   * Checks if this Spinner's date time is equal to the given date time.
301   *
302   * @param otherLocalTime the other local time
303   * @return true, if is equal
304   */
305  public boolean isEqual(final LocalDateTime otherLocalTime) {
306    return compareDateTime(otherLocalTime) == 0;
307  }  
308  
309  /**
310   * Checks if this Spinner's date time is after the given date time.
311   *
312   * @param otherLocalTime the other local time
313   * @return true, if is after
314   */
315  public boolean isAfter(final LocalDateTime otherLocalTime) {
316    return compareDateTime(otherLocalTime) > 0;
317  }  
318  
319  /**
320   * Compare this Spinner's date time to the given date time.
321   *
322   * @param otherLocalTime the other local time
323   * @return the int
324   */
325  private int compareDateTime(final LocalDateTime otherLocalTime) {
326    final LocalDateTime meDt = getValue().atDate(LocalDate.now());
327    return meDt.isAfter(otherLocalTime) ? 1 : meDt.isBefore(otherLocalTime) ? -1 : 0;
328  }
329
330  /**
331   * @see java.lang.Comparable#compareTo(java.lang.Object)
332   */
333  @Override
334  public int compareTo(final Spinner<LocalTime> other) {
335    final LocalDateTime otherDt = other.getValue().atDate(LocalDate.now());
336    return isAfter(otherDt) ? 1 : isEqual(otherDt) ? 0 : -1;
337  }
